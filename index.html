    <!-- JavaScript for fetching data and animating the total -->

    <script>

        // ====================================================================================

        // The placeholder URL has been replaced with your live Google Sheet link.

        // ====================================================================================

        const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSPeAHz8TtmdotclMBHxEt22hn5-KdII-9Fh_hy636XUcxqmSjN5f0lfYhrGP87MXvsU03ThAAWMsck/pub?gid=0&single=true&output=csv';

        

        function animateValue(obj, start, end, duration) {

            let startTimestamp = null;

            const step = (timestamp) => {

                if (!startTimestamp) startTimestamp = timestamp;

                const progress = Math.min((timestamp - startTimestamp) / duration, 1);

                const currentValue = Math.floor(progress * (end - start) + start);

                

                obj.innerHTML = new Intl.NumberFormat('en-US', {

                    style: 'currency',

                    currency: 'USD',

                    minimumFractionDigits: 0,

                    maximumFractionDigits: 0

                }).format(currentValue);


                if (progress < 1) {

                    window.requestAnimationFrame(step);

                } else {

                     obj.innerHTML = new Intl.NumberFormat('en-US', {

                        style: 'currency',

                        currency: 'USD'

                    }).format(end);

                }

            };

            window.requestAnimationFrame(step);

        }


        function fetchData() {

            const totalElement = document.getElementById('total-raised');

            

            fetch(sheetUrl)

                .then(response => {

                    if (!response.ok) throw new Error('Network response was not ok');

                    return response.text();

                })

                .then(csvText => {

                    const rows = csvText.split('\n').slice(1);

                    let total = 0;

                    rows.forEach(row => {

                        const columns = row.split(',');

                        // IMPORTANT: Assumes the 'Amount' is in the 3rd column (index 2).

                        if (columns.length > 2) {

                            const amount = parseFloat(columns[2]);

                            if (!isNaN(amount)) total += amount;

                        }

                    });

                    

                    animateValue(totalElement, 0, total, 1500);

                })

                .catch(error => {

                    console.error('Error fetching or processing data:', error);

                    totalElement.innerText = '$0.00';

                });

        }


        document.addEventListener('DOMContentLoaded', fetchData);

    </script>
